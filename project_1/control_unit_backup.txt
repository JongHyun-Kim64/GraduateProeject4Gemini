module control_unit(
    input clk,
    input rst,
    output reg test_sig,
    output reg shift_sig
    );
    
    reg [3:0] clkcnt;
    
    always @(posedge clk, posedge rst) begin
        if(rst) begin
            clkcnt <= 0;
            test_sig <= 0;
            shift_sig <= 0;
        end else begin
            clkcnt <= clkcnt + 1;
            case(clkcnt)
                5: shift_sig <= 1;
                10: begin
                    test_sig <= 1;
                    clkcnt <=0;
                end
                default: begin
                    test_sig <= 0;
                    shift_sig <= 0;
                end
            endcase
        end
    end
endmodule
